function submitJob() { let company = document.getElementById('job_company').value.trim(); if (!company) { showModal('事業者名を入力してください。'); return; } let candInputDivs = document.getElementById('job_cand_list').children; let candidates = []; Array.from(candInputDivs).forEach(div => { let inputEl = div.querySelector('input.job-cand-id'); let nameEl = div.querySelector('span.cand-name-display'); if (inputEl && inputEl.value.trim()) { let idStr = formatId(inputEl.value.trim()); let nameStr = nameEl ? nameEl.innerText.trim() : ""; if (nameStr && nameStr !== "該当なし" && nameStr !== "読込中...") { candidates.push(`${idStr}-${nameStr}`); } else { candidates.push(idStr); } } }); if (candidates.length === 0) { showModal('候補者を少なくとも1名入力してください。'); return; } let fileInputs = document.querySelectorAll('.job-file-input'); let relatedFiles = []; fileInputs.forEach(input => { let val = input.value.trim(); if(val) relatedFiles.push(val); }); let formData = { company: company, skill: document.getElementById('job_skill').value, candidates: candidates, interviewDate: document.getElementById('job_interviewDate').value, relatedFiles: relatedFiles, memo: document.getElementById('job_memo').value }; document.getElementById('jobBtn').disabled = true; showModal('案件を登録中...', true); google.script.run.withSuccessHandler(function(result) { document.getElementById('jobBtn').disabled = false; // 入力項目のクリア document.getElementById('job_company').value = ''; document.getElementById('job_skill').value = ''; document.getElementById('job_interviewDate').value = ''; document.getElementById('job_file_list').innerHTML = ''; addJobFileRow('job'); document.getElementById('job_memo').value = ''; document.getElementById('job_cand_list').innerHTML = createCandRowHTML('job-cand-id'); if (result.isNewCompany) { closeModal(); // 新規事業者の場合の特別案内 if (confirm(result.message + '\n\n新しい事業者『' + result.companyName + '』をマスタに仮登録しました。続けて住所やHPなどの詳細情報を登録しますか？')) { // 事業者マスタ登録画面へ遷移（名前を引き継ぐ） google.script.run.showSidebarCompany(result.companyName); } } else { showModal(result.message); } }).addJob(formData); }
